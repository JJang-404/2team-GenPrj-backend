import torch
from diffusers import StableDiffusion35Pipeliner
from fastapi import FastAPI
import uvicorn

app = FastAPI()

# 1. 모델 로드 (L4 VRAM의 약 70% 수준인 fp16 사용)
model_id = "stabilityai/stable-diffusion-3.5-large"
pipe = StableDiffusion35Pipeliner.from_pretrained(model_id, torch_dtype=torch.float16)
pipe.to("cuda")

# 2. Triton 가속 (남은 30%를 연산 버퍼로 활용)
pipe.transformer = torch.compile(pipe.transformer, mode="reduce-overhead")

@app.post("/generate")
async def generate(prompt: str):
    image = pipe(prompt=prompt, num_inference_steps=30).images[0]
    # 이미지 처리 로직...
    return {"status": "success"}

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8001)

import torch
from diffusers import StableDiffusion3Pipeline

# 1. 모델 ID 설정 (Medium 버전)
model_id = "stabilityai/stable-diffusion-3.5-medium"

print("🚀 SD 3.5 Medium 모델 로드 중... (Large보다 훨씬 가볍습니다)")

# 2. 모델 로드 (VRAM 16GB 최적화 세팅)
pipe = StableDiffusion3Pipeline.from_pretrained(
    model_id, 
    torch_dtype=torch.bfloat16
)

# 3. 5060 Ti를 위한 메모리 관리
pipe.enable_model_cpu_offload() 
# 만약 속도를 더 높이고 싶고 VRAM 여유가 있다면 위 라인 대신 pipe.to("cuda") 사용

# 4. 이미지 생성 테스트
prompt = "A high-quality advertisement poster for a cozy cafe, delicious latte art on a wooden table, warm morning sunlight, cinematic 8k"

print("✨ 이미지 생성 시작...")
with torch.inference_mode():
    image = pipe(
        prompt, 
        num_inference_steps=30, 
        guidance_scale=4.5 # Medium은 Large보다 낮은 guidance scale을 권장하기도 합니다
    ).images[0]

# 5. 저장
image.save("result_sd35_medium.png")
print("🎉 성공! result_sd35_medium.png 파일을 확인하세요.")

@router.post("/changeimage")
async def changeimage(req: ChangeImageRequest):
    if req.strength < 0.0 or req.strength > 1.0:
        raise HTTPException(status_code=400, detail="strength는 0.0~1.0 범위여야 합니다.")

    print(f"🛠️ 이미지 변환 요청 프롬프트: {req.prompt}")

    raw_base64 = req.image_base64
    if "," in raw_base64:
        raw_base64 = raw_base64.split(",", 1)[1]

    try:
        input_bytes = base64.b64decode(raw_base64, validate=True)
        init_image = Image.open(BytesIO(input_bytes)).convert("RGB")
    except Exception as exc:
        raise HTTPException(status_code=400, detail="유효한 base64 이미지가 아닙니다.") from exc

    image = diffusion_service.change_image(
        prompt=req.prompt,
        init_image=init_image,
        strength=req.strength,
    )

    img_byte_arr = BytesIO()
    image.save(img_byte_arr, format="PNG")

    print("✅ 이미지 변환 완료!")
    return Response(content=img_byte_arr.getvalue(), media_type="image/png")
